class Solution {
public:
    void sortColors(vector<int>& nums) {
        int n=nums.size();
        for(int r=1;r<=n-1;r++){
            for(int i=0;i<=n-r-1;i++){
                if(nums[i+1]<nums[i]){
                    swap(nums[i+1],nums[i]);
                }
            }
        }
    }
};
