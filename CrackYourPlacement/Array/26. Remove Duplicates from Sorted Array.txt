class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int length=nums.size()-1;
        int slow=0;
        int fast=0;
        while(fast<=length){
            if(nums[slow]==nums[fast])
                fast++;
            else{
                slow++;
                swap(nums[slow],nums[fast]);
                fast++;
            }
            }
             return slow+1;
            }
        
};
