public:
    long long findMinDiff(vector<long long> a, long long n, long long m){
    //code 
           int min=INT_MAX;
           sort(a.begin(),a.end());
           for(int i=0;i<=n-m;i++){
               if(min>abs(a[i]-a[i+m-1]))
                  min=abs(a[i]-a[i+m-1]);
           }
           return min;
    }
